ℰ : Type a → Type a
ℰ A = Σ[ xs ⦂ Stream A ] Π[ x ⦂ A ] ∥ x ∈ xs ∥

ℰ≡ℕ↠ : ℰ A ≡ (ℕ ↠ A)
ℰ≡ℕ↠ = refl

_∥Σ∥_ : ℰ A → (∀ x → ℰ (U x)) → ℰ (Σ A U)
(xs ∥Σ∥ ys) .fst = cantor (xs .fst) (fst ∘ ys)
