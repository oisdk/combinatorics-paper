  ℰ⇒ℰ! _≟_ E .fst = E .fst
  ℰ⇒ℰ! _≟_ E .snd x = recompute ((_≟ x) ∈? E .fst) (E .snd x)

  ℰ!⇒ℰ : ℰ! A → ℰ A
  ℰ!⇒ℰ E .fst = E .fst
  ℰ!⇒ℰ E .snd x = ∣ E .snd x ∣

  _∥Σ∥_ : {B : A → Type b} → ℰ A → ((x : A) → ℰ (B x)) → ℰ (Σ A B)
  (xs ∥Σ∥ ys) .fst = sup-Σ (xs .fst) (fst ∘ ys)
  (xs ∥Σ∥ ys) .snd (x , y) = ⦇ (cov-Σ x y (xs .fst) (fst ∘ ys)) (xs .snd x) (ys x .snd y) ⦈
